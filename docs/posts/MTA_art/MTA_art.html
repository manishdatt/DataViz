 basefmt=' ', bottom=34)

plt.setp(ml,  marker="*")
for i, txt in enumerate(material_timeline['art_material']):
    ax.annotate(txt, (material_timeline['art_date'][i]+0.5, material_timeline['art_material'][i]), xytext=(0, -2), textcoords='offset points')
